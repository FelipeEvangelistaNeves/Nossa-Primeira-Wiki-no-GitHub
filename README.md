# Lista de exercícios 


## 1. ls

Função:
Lista os arquivos e pastas dentro de um diretório.

Exemplo:

bash
ls -l




## 2. cd

Função:
Permite mudar de pasta (diretório).

Exemplo:

bash
cd Documentos




## 3. pwd

Função:
Mostra o caminho completo da pasta atual.

Exemplo:

bash
pwd




## 4. mkdir

Função:
Cria uma nova pasta.

Exemplo:

bash
mkdir projetos




## 5. rmdir

Função:
Remove uma pasta vazia.

Exemplo:

bash
rmdir projetos




## 6. touch

Função:
Cria um arquivo vazio.

Exemplo:

bash
touch arquivo.txt




## 7. cp

Função:
Copia arquivos ou pastas.

Exemplo:

bash
cp arquivo.txt copia.txt




## 8. mv

Função:
Move ou renomeia arquivos e pastas.

Exemplo:

bash
mv arquivo.txt novo_nome.txt




## 9. rm

Função:
Remove arquivos ou pastas.

Exemplo:

bash
rm arquivo.txt




## 10. find

Função:
Procura arquivos dentro de diretórios.

Exemplo:

bash
find . -name "arquivo.txt"




## 11. cat

Função:
Mostra o conteúdo de um arquivo.

Exemplo:

bash
cat arquivo.txt




## 12. less

Função:
Permite visualizar arquivos grandes página por página.

Exemplo:

bash
less arquivo.txt




## 13. head

Função:
Mostra as primeiras linhas de um arquivo.

Exemplo:

bash
head arquivo.txt




## 14. tail

Função:
Mostra as últimas linhas de um arquivo.

Exemplo:

bash
tail arquivo.txt




## 15. nano

Função:
Editor de texto simples no terminal.

Exemplo:

bash
nano arquivo.txt




## 16. grep

Função:
Busca palavras dentro de arquivos.

Exemplo:

bash
grep "erro" arquivo.txt




## 17. wc

Função:
Conta linhas, palavras e caracteres de um arquivo.

Exemplo:

bash
wc arquivo.txt




## 18. sort

Função:
Organiza o conteúdo em ordem.

Exemplo:

bash
sort lista.txt




## 19. uniq

Função:
Remove linhas repetidas.

Exemplo:

bash
uniq lista.txt




## 20. top

Função:
Mostra os processos em execução em tempo real.

Exemplo:

bash
top




## 21. htop

Função:
Versão mais visual do top.

Exemplo:

bash
htop




## 22. ps

Função:
Lista processos em execução.

Exemplo:

bash
ps aux




## 23. kill

Função:
Finaliza um processo.

Exemplo:

bash
kill 1234




## 24. chmod

Função:
Altera permissões de arquivos.

Exemplo:

bash
chmod 755 arquivo.sh




## 25. chown

Função:
Altera o dono de um arquivo.

Exemplo:

bash
chown usuario arquivo.txt




## 26. ping

Função:
Testa conexão com outro servidor.

Exemplo:

bash
ping google.com




## 27. ifconfig

Função:
Mostra informações de rede.

Exemplo:

bash
ifconfig




## 28. wget

Função:
Baixa arquivos da internet.

Exemplo:

bash
wget https://site.com/arquivo.zip




## 29. curl

Função:
Faz requisições para URLs.

Exemplo:

bash
curl https://api.site.com




## 30. man

Função:
Mostra o manual de um comando.

Exemplo:

bash
man ls
